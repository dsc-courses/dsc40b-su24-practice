\begin{prob}
    Suppose $f_1(n)$ is  $O(n^2)$ and $\Omega(n)$. Also suppose that $f_2(n) = \Theta(n^2)$.

    Consider the function $g(n) = f_2(n) / f_1(n)$. True or false: it must be 
    the case that $g(n) = \Omega(n)$.
    \tF{}

    \begin{soln}
        False.
    \end{soln}
\end{prob}

\begin{prob}
    Let
    \[
        f(n) = 
        \frac{n^3 - 2n + 100}{\sqrt n}
        \cdot
        \frac{n}{n^3 + n^2 + \sin n}
        \cdot n^2
    \]

    True or False: $f(n) = \Theta(n^2)$.
    \tF{}

    \begin{soln}
        False.
    \end{soln}

\end{prob}

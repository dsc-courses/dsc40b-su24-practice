\begin{prob}
    Solve the recurrence below, stating the solution in asymptotic notation. Show your work.

    \[
        T(n) =
        \begin{cases}
            2 T(n/4) + \Theta(n),& n > 1\\
            1,& n = 1
        \end{cases}
    \]

    \begin{soln}
        $\Theta(n)$
    \end{soln}

\end{prob}

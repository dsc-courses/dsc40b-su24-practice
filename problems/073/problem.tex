\begin{prob}
    Let
    \[
        f(n) = 5 n \log n +
        \frac{n^3 + 5}{n + 2 + |\sin \pi n|}
        +
        n \sqrt n
    \]

    Write $f$ in asymptotic notation in as simplest terms possible.

    \begin{soln}
        $f(n) = \Theta(n^2)$
    \end{soln}

\end{prob}

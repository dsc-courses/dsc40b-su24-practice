\begin{prob}
    What is the time complexity of the following function?
    State your answer using asymptotic notation (e.g., $\Theta(n)$).

    \begin{minted}{python}
            def foo(n):
                for i in range(n):
                    for j in range(n):
                        for k in range(n**2):
                            print(i + j + k)
    \end{minted}

    \begin{soln}
        $\Theta(n^4)$
    \end{soln}

\end{prob}

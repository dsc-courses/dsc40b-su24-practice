\begin{prob}
    Suppose $f_1(n) = \Omega(g_1(n))$ and $f_2 = \Omega(g_2(n))$.
    Define $f(n) = \min \{ f_1(n), f_2(n) \}$ and $g(n) = \min \{ g_1(n), g_2(n)\}$.

    True or false: it is necessarily the case that $f(n) = \Omega(g(n))$.

     \Tf{}

     \begin{soln}
         True.
     \end{soln}
\end{prob}

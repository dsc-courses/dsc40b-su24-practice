\begin{prob}
    Suppose $f_1(n)$ is $O(n^2)$ and $\Omega(n)$. Also suppose that $f_2(n) = \Theta(n^2)$.

    Consider the function $f(n) = f_1(n) + f_2(n)$. True or false: it must be 
    the case that $f(n) = \Theta(n^2)$.

    \Tf{}

    \begin{soln}
        True.
    \end{soln}
\end{prob}

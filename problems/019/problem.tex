\begin{prob}

    Suppose \mintinline{python}{mergesort} is called on an array of size $n$. In total, how
    many calls to \mintinline{python}{merge} are made over all calls to \mintinline{python}{mergesort}?

    \begin{choices}
        \choice $\Theta(\log n)$
        \choice $\Theta(n)$
        \choice $\Theta(n \log n)$
        \choice $\Theta(n^2)$
    \end{choices}

    \begin{soln}
        $\Theta(n)$
    \end{soln}

\end{prob}

\begin{prob}
    Suppose \mintinline{python}{binary_search} is called on a sorted array of
    size $n$. In total, how many recursive calls to
    \mintinline{python}{binary_search} are made in the worst case?

    \begin{choices}
        \correctchoice $\Theta(\log n)$
        \choice $\Theta(n)$
        \choice $\Theta(n \log n)$
        \choice $\Theta(n^2)$
    \end{choices}

    \begin{soln}
        $\Theta(\log n)$
    \end{soln}

\end{prob}

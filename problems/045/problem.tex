\begin{prob}
    Suppose Algorithm A takes $\Theta(n^2)$ time, while Algorithm B takes
    $\Theta(2^n)$ time.
    True or False: there \textbf{could} exist an input on which
    Algorithm $B$ takes less time than Algorithm A.

        \Tf{}

        \begin{soln}
            True.
        \end{soln}

\end{prob}

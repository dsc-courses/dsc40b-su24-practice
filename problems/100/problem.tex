\begin{prob}
    An undirected graph with 20 nodes and 30 edges has three connected components:
    $A$, $B$, and $C$. Connected component $A$ has 7 nodes. What is the smallest
    number of edges it can have?

    \begin{soln}
        6
    \end{soln}

\end{prob}
